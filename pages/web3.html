const immDomainsAbi = [...]
const immDomainsAddress = '0xeb5d45f2c9e15cbd4885f98a4fb59623077a1dc4'
const immDomains = web3.eth.contract(immDomainsAbi).at(immDomainsAddress)

const nullAddress = '0x0000000000000000000000000000000000000000'

function getAddress(callback) {
  return immDomains.addresses(web3.fromAscii('weth'), (error, address) => {
    if (error) {
      callback(error, null)
    } else {
      if (address === nullAddress) {
        callback(null, null)
      } else {
        callback(null, address)
      }
    }
  })
}
